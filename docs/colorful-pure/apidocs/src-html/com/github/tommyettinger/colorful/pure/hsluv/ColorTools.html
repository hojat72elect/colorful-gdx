035/**
036 * Contains code for manipulating colors as {@code int} and packed {@code float} values in the HSLuv color space.
037 * See <a href="https://www.hsluv.org/">HSLuv's website</a> for more info.
038 * <br>
039 * The HSLuv color space has 3 channels, Hue, Saturation, and Lightness. Lightness should be much more even when hue and
040 * saturation change, when compared with "vanilla" HSL or HSV. This also has an alpha channel in each color,
041 * which acts like it does in every other color space package here (multiplicative alpha, higher is more opaque).
042 */

049    /**
050     * Gets a packed float representation of a color given as 4 float components, here, hue, saturation, lightness, and
051     * alpha (or opacity). H should be between 0 and 1, inclusive, with 0.0f or so meaning reddish, 0.3f or so meaning
052     * greenish, and 0.7 or so meaning bluish. S should be between 0 and 1, inclusive, with 0 meaning grayscale and 1
053     * meaning fully bold/bright. L should be between 0 and 1, inclusive, with 0 used for very dark colors (almost only
054     * black), and 1 used for very light colors (almost only white). Alpha is the multiplicative opacity of the color,
055     * and acts like RGBA's alpha.
056     *
057     * @param h     0f to 1f, hue component, with 0.0f meaning red, 0.3 meaning green, and 0.7 meaning blue
058     * @param s     0f to 1f, saturation component, with 0.0f meaning grayscale and 1.0f meaning fully bold/bright
059     * @param l     0f to 1f, lightness component, with 0.5f meaning "no change" and 1f brightening
060     * @param alpha 0f to 1f, 0f makes the color transparent and 1f makes it opaque
061     * @return a float encoding a color with the given properties
062     */
063    public static float hsluv(float h, float s, float l, float alpha) {
064        return BitConversion.intBitsToFloat(((int) (alpha * 255.999f) << 24 & 0xFE000000) | ((int) (l * 255.999f) << 16 & 0xFF0000)
065                | ((int) (s * 255.999f) << 8 & 0xFF00) | ((int) (h * 255.999f) & 0xFF));
066    }
067    /**
068     * Gets a packed float representation of a color given as 4 float components, H, S, L, and alpha, with the S, L, and
069     * alpha components clamped to the 0f to 1f range before being entered into the packed float color, while H is
070     * wrapped into the same range. This is only different from {@link #hsluv(float, float, float, float)} in that it
071     * clamps or wraps each component.
072     *
073     * @see #hsluv(float, float, float, float) This uses the same definitions for H, S, L, and alpha as hsluv().
074     * @param h     0f to 1f, hue component, with 0.0f meaning red, 0.3 meaning green, and 0.7 meaning blue
075     * @param s     0f to 1f, saturation component, with 0.0f meaning grayscale and 1.0f meaning fully bold/bright
076     * @param l     0f to 1f, lightness component, with 0.5f meaning "no change" and 1f brightening
077     * @param alpha 0f to 1f, 0f makes the color transparent and 1f makes it opaque
078     * @return a float encoding a color with the given properties
079     */
080    public static float clamp(float h, float s, float l, float alpha) {
081        return BitConversion.intBitsToFloat((Math.min(Math.max((int) (alpha * 127.999f), 0), 127) << 25)
082                | (Math.min(Math.max((int) (l * 255.999f), 0), 255) << 16)
083                | (Math.min(Math.max((int) (s * 255.999f), 0), 255) << 8)
084                | (int) ((h - floor(h)) * 256f)
085        );
086    }

088    /**
089     * An approximation of the cube-root function for float inputs and outputs.
090     * This can be about twice as fast as {@link Math#cbrt(double)}. This
091     * version does not tolerate negative inputs, because in the narrow use
092     * case it has in this class, it never is given negative inputs.
093     * <br>
094     * Has very low relative error (less than 1E-9) when inputs are uniformly
095     * distributed between 0 and 512, and absolute mean error of less than
096     * 1E-6 in the same scenario. Uses a bit-twiddling method similar to one
097     * presented in Hacker's Delight and also used in early 3D graphics (see
098     * https://en.wikipedia.org/wiki/Fast_inverse_square_root for more, but
099     * this code approximates cbrt(x) and not 1/sqrt(x)). This specific code
100     * was originally by Marc B. Reynolds, posted in his "Stand-alone-junk"
101     * repo: https://github.com/Marc-B-Reynolds/Stand-alone-junk/blob/master/src/Posts/ballcube.c#L182-L197 .
102     * It's worth noting that while hardware instructions for finding the
103     * square root of a float have gotten extremely fast, the same is not
104     * true for the cube root (which has to allow negative inputs), so while
105     * the bit-twiddling inverse square root is no longer a beneficial
106     * optimization on current hardware, this does seem to help.
107     * <br>
108     * This is used when converting from RGB to HSLuv, as an intermediate step.
109     * @param x any non-negative finite float to find the cube root of
110     * @return the cube root of x, approximated
111     */
112    private static float cbrtPositive(float x) {
113        int ix = BitConversion.floatToRawIntBits(x);
114        final float x0 = x;
115        ix = (ix>>>2) + (ix>>>4);
116        ix += (ix>>>4);
117        ix += (ix>>>8) + 0x2A5137A0;
118        x  = BitConversion.intBitsToFloat(ix);
119        x  = 0.33333334f*(2f * x + x0/(x*x));
120        x  = 0.33333334f*(2f * x + x0/(x*x));
121        return x;
122    }

134     * Used when given non-linear sRGB inputs to make them linear, using an exact gamma of 2.4 and accounting for the
135     * darkest colors with a different formula. This is rather close to squaring {@code component}.
136     * @param component any non-linear channel of a color, to be made linear
137     * @return a linear version of component
138     */
139    private static float forwardGamma(final float component) {
140        return component < 0.04045f ? component * (1f/12.92f) : (float)Math.pow((component + 0.055f) * (1f/1.055f), 2.4f);
141    }

143    /**
144     * Used to return from a linear, gamma-corrected input to an sRGB, non-linear output, using an exact gamma of 2.4
145     * and accounting for the darkest colors with a different formula. This is mostly similar to the square root, but is
146     * more precise for very dark colors.
147     * @param component a linear channel of a color, to be made non-linear
148     * @return a non-linear version of component
149     */
150    private static float reverseGamma(final float component) {
151        return component < 0.0031308f ? component * 12.92f : (float)Math.pow(component, 1f/2.4f) * 1.055f - 0.055f;
152    }

154    /**
155     * Changes the curve of a requested L value so that it matches the internally-used curve. This takes a curve with a
156     * dark area similar to sRGB (a fairly small one), and makes it significantly larger. This is typically used on
157     * "to HSLuv" conversions. This is much less potent of a change than the method used by Oklab.
158     * @param L lightness, from 0 to 1 inclusive
159     * @return an adjusted L value that can be used internally
160     */
161    public static float forwardLight(final float L) {
162        final float shape = 0.8528f, turning = 0.1f;
163        final float d = turning - L;
164        if(d < 0)
165            return ((1f - turning) * (L - 1f)) / (1f - (L + shape * d)) + 1f;
166        else
167            return (turning * L) / (1e-20f + (L + shape * d));
168    }
169    /**
170     * Changes the curve of the internally-used lightness when it is output to another format. This makes the dark area
171     * area smaller, matching (kind-of) the curve that the standard sRGB lightness uses. This is typically used on "from
172     * HSLuv" conversions. This is much less potent of a change than the method used by Oklab.
173     * @param L lightness, from 0 to 1 inclusive
174     * @return an adjusted L value that can be fed into a conversion to RGBA or something similar
175     */
176    public static float reverseLight(final float L) {
177        final float shape = 1.1726f, turning = 0.1f;
178        final float d = turning - L;
179        if(d < 0)
180            return ((1f - turning) * (L - 1f)) / (1f - (L + shape * d)) + 1f;
181        else
182            return (turning * L) / (1e-20f + (L + shape * d));
183    }

198    /**
199     * Converts a packed float color in the format produced by {@link #hsluv(float, float, float, float)} to an RGBA8888 int.
200     * This format of int can be used with Pixmap and in some other places in libGDX.
201     * @param packed a packed float color, as produced by {@link #hsluv(float, float, float, float)}
202     * @return an RGBA8888 int color
203     */
204    public static int toRGBA8888(final float packed)
205    {
206        final int decoded = BitConversion.floatToRawIntBits(packed);
207        float H = ((decoded & 0xff) / 255f);
208        float S = ((decoded >>> 8 & 0xff) / 255f);
209        float L = reverseLight((decoded >>> 16 & 0xff) / 255f);
210
211        // HSLuv to Lch
212        float C;
213        if (L > 0.99999f) {
214            L = 1;
215            C = 0;
216        } else if (L < 0.00001f) {
217            L = 0;
218            C = 0;
219        } else
220            C = chromaLimit(H, L) * S;
221
222        // Lch to Luv
223        float U = TrigTools.cosTurns(H) * C;
224        float V = MathTools.sin_(H) * C;
225
226        // Luv to XYZ
227        float x, y, z;
228        if (L < 0.00001f) {
229            return (decoded & 0xfe000000) >>> 24 | decoded >>> 31;
230        } else if(L > 0.9999f) {
231            return 0xFFFFFF00 | (decoded & 0xfe000000) >>> 24 | decoded >>> 31;
232        }else {
233            if (L <= 0.08f)
234                y = L / kappa;
235            else {
236                y = (L + 0.16f) / 1.16f;
237                y *= y * y;
238            }
239            float iL = 1f / (13f * L);
240            float varU = U * iL + refU;
241            float varV = V * iL + refV;
242            x = 9 * varU * y / (4 * varV);
243            z = (3 * y / varV) - x / 3 - 5 * y;
244        }
245        final int r = (int)(reverseGamma(+3.2404542f * x + -1.5371385f * y + -0.4985314f * z) * 255.999f);
246        final int g = (int)(reverseGamma(-0.9692660f * x + +1.8760108f * y + +0.0415560f * z) * 255.999f);
247        final int b = (int)(reverseGamma(+0.0556434f * x + -0.2040259f * y + +1.0572252f * z) * 255.999f);
248        return r << 24 | g << 16 | b << 8 | (decoded & 0xfe000000) >>> 24 | decoded >>> 31;
249    }

305    /**
306     * Takes a color encoded as an RGBA8888 int and converts to a packed float in the HSLuv format this uses.
307     * @param rgba an int with the channels (in order) red, green, blue, alpha; should have 8 bits per channel
308     * @return a packed float as HSLuv, which this class can use
309     */
310    public static float fromRGBA8888(final int rgba) {
311        final float r = forwardGamma((rgba >>> 24) * 0x1.010101010101p-8f);
312        final float g = forwardGamma((rgba >>> 16 & 0xFF) * 0x1.010101010101p-8f);
313        final float b = forwardGamma((rgba >>> 8 & 0xFF) * 0x1.010101010101p-8f);
314
315        final float x = /* forwardXYZ */(0.4124564f * r  + 0.3575761f * g + 0.1804375f * b);
316        final float y = /* forwardXYZ */(0.2126729f * r  + 0.7151522f * g + 0.0721750f * b);
317        final float z = /* forwardXYZ */(0.0193339f * r  + 0.1191920f * g + 0.9503041f * b);
318
319        // XYZ to Luv
320        float L = 1.16f * cbrtPositive(y) - 0.16f, U, V, h, s, l;
321        if (L < 0.00001f) {
322            L = 0;
323            U = 0;
324            V = 0;
325        } else {
326            U = 13 * L * (4 * x / (x + 15 * y + 3 * z) - refU);
327            V = 13 * L * (9 * y / (x + 15 * y + 3 * z) - refV);
328        }
329
330        // Luv to Lch
331        float C = (float)Math.sqrt(U * U + V * V);
332        h = MathTools.atan2_(V, U);
333
334        // Lch to HSLuv
335        if (L > 0.99999f) {
336            s = 0;
337            l = 1;
338        } else if (L < 0.00001f) {
339            s = 0;
340            l = 0;
341        } else {
342            l = forwardLight(L);
343            s = Math.min(C / chromaLimit(h, l), 1);
344        }
345        return BitConversion.intBitsToFloat(
346                          Math.min(Math.max((int)(h * 255.999f    ), 0), 255)
347                        | Math.min(Math.max((int)(s * 255.999f    ), 0), 255) << 8
348                        | Math.min(Math.max((int)(l * 255.999f    ), 0), 255) << 16
349                        | (rgba & 0xFE) << 24);
350    }

769     * values have 0 saturation and 0 chroma, while brighter colors have high saturation and chroma. The difference is

793    /**
794     * Given a hue and lightness, this gets the (exact) maximum chroma possible for that hue-lightness
795     * combination, using HSLuv's versions of lightness and hue (not raw HSL). This is useful to know the bounds of
796     * {@link #chroma(float)}. This should be no greater than 1.0f . Note that this version of chromaLimit() is
797     * a little slower than Oklab's version, because this has to go to greater lengths to become accurate.
798     * @param hue the hue, typically between 0.0f and 1.0f, to look up
799     * @param lightness the lightness, clamped between 0.0f and 1.0f, to look up
800     * @return the maximum possible chroma for the given hue and lightness, between 0.0f and 1.0f
801     */
802    public static float chromaLimit(final float hue, final float lightness) {
803        final float h = hue - floor(hue);
804        float sin = MathTools.sin_(h);
805        float cos = TrigTools.cosTurns(h);
806        float sub1 = (lightness + 0.16f) / 1.16f;
807        sub1 *= sub1 * sub1;
808        float sub2 = sub1 > epsilon ? sub1 : lightness / kappa;
809        float min = Float.MAX_VALUE;
810        for (int i = 0; i < 3; i++) {
811            float m1 = m[i][0] * sub2, m2 = m[i][1] * sub2, m3 = m[i][2] * sub2;
812            for (int t = 0; t < 2; t++) {
813                m2 -= t;
814                float top1 = 2845.17f * m1 - 948.39f * m3;
815                float top2 = (8384.22f * m3 + 7698.60f * m2 + 7317.18f * m1) * lightness;
816                float bottom = (6322.60f * m3 - 1264.52f * m2);
817                float length = intersectLength(sin, cos, top1 / bottom, top2 / bottom);
818                if (length >= 0) min = Math.min(min, length);
819            }
820        }
821        return min;
822    }

1289        float S = (decoded >>> 8 & 0xff) / 255f;

1291        float alpha = (decoded >>> 25) / 127f;
1292

1582    /**
1583     * Makes a quasi-randomly-edited variant on the given {@code color}, allowing typically a small amount of
1584     * {@code variance} (such as 0.05 to 0.25) between the given color and what this can return. The {@code seed} should
1585     * be different each time this is called, and can be obtained from a random number generator to make the colors more
1586     * random, or can be incremented on each call. If the seed is only incremented or decremented, then this shouldn't
1587     * produce two similar colors in a row unless variance is very small. The variance affects the H, S, and L of the
1588     * generated color, and each of those channels can go up or down by the given variance as long as the total distance
1589     * isn't greater than the variance.
1590     * @param color a packed float color, as produced by {@link #hsluv(float, float, float, float)}
1591     * @param seed a long seed that should be different on each call; should not be 0
1592     * @param variance max amount of difference between the given color and the generated color; always less than 1
1593     * @return a generated packed float color that should be at least somewhat different from {@code color}
1594     */
1595    public static float randomEdit(final float color, long seed, final float variance) {
1596        final int decoded = BitConversion.floatToRawIntBits(color);
1597        float H = (decoded & 0xff) / 255f;
1598        float S = (decoded >>> 8 & 0xff) / 255f;
1599        float L = reverseLight((decoded >>> 16 & 0xff) / 255f);
1600        final float limit = variance * variance;
1601        float dist, x, y, z;
1602        for (int j = 0; j < 50; j++) {
1603            x = (((seed * 0xD1B54A32D192ED03L >>> 41) - 0x7FFFFFp-1f) * 0x1p-22f) * variance;
1604            y = (((seed * 0xABC98388FB8FAC03L >>> 41) - 0x7FFFFFp-1f) * 0x1p-22f) * variance;
1605            z = (((seed * 0x8CB92BA72F3D8DD7L >>> 41) - 0x7FFFFFp-1f) * 0x1p-22f) * variance;
1606            seed += 0x9E3779B97F4A7C15L;
1607            dist = x * x + y * y + z * z;
1608            if(dist <= limit)
1609                return clamp(H + x, S + y, L + z, (decoded >>> 25) / 127f);
1610        }
1611        return color;
1612    }

1636    /**
1637     * Interpolates from the packed float color start towards end by change. Both start and end should be packed colors,
1638     * as from {@link #hsluv(float, float, float, float)}, and change can be between 0f (keep start) and 1f (only use
1639     * end). Both start and end must use HSLuv. This is a good way to reduce allocations of temporary Colors.
1640     * @param start the starting color as a packed float
1641     * @param end the target color as a packed float
1642     * @param change how much to go from start toward end, as a float between 0 and 1; higher means closer to end
1643     * @return a packed float that represents a color between start and end
1644     */
1645    public static float lerpFloatColors(final float start, final float end, float change) {
1646        final int s = BitConversion.floatToRawIntBits(start), e = BitConversion.floatToRawIntBits(end),
1647                hs = (s & 0xFF), ss = (s >>> 8) & 0xFF, ls = (s >>> 16) & 0xFF, as = s >>> 24 & 0xFE,
1648                he = (e & 0xFF), se = (e >>> 8) & 0xFF, le = (e >>> 16) & 0xFF, ae = e >>> 24 & 0xFE;
1649        float Hs = hs / 255f, Cs, Ls;
1650        if (ls == 255) {
1651            Ls = 1f;
1652            Cs = 0f;
1653        } else if (ls == 0) {
1654            Ls = 0f;
1655            Cs = 0f;
1656        } else {
1657            Ls = ls / 255f;
1658            Cs = chromaLimit(Hs, Ls) * (ss / 255f);
1659        }
1660        // Lch to Luv
1661        float Us = TrigTools.cosTurns(Hs) * Cs;
1662        float Vs = MathTools.sin_(Hs) * Cs;
1663
1664        float He = he / 255f, Ce, Le;
1665        if (le == 255) {
1666            Le = 1f;
1667            Ce = 0f;
1668        } else if (le == 0) {
1669            Le = 0f;
1670            Ce = 0f;
1671        } else {
1672            Le = le / 255f;
1673            Ce = chromaLimit(He, Le) * (se / 255f);
1674        }
1675        // Lch to Luv
1676        float Ue = TrigTools.cosTurns(He) * Ce;
1677        float Ve = TrigTools.sinTurns(He) * Ce;
1678
1679        float H, S, L, U, V;
1680
1681        L = (Ls + change * (Le - Ls));
1682        U = (Us + change * (Ue - Us));
1683        V = (Vs + change * (Ve - Vs));
1684
1685        // Luv to Lch
1686        H = MathTools.atan2_(V, U);
1687
1688
1689        // Lch to HSLuv
1690        if (L > 0.99999f) {
1691            return Palette.WHITE;
1692        } else if (L < 0.00001f) {
1693            return Palette.BLACK;
1694        } else {
1695            S = Math.min((float)Math.sqrt(U * U + V * V) / chromaLimit(H, L), 1);
1696        }
1697        return BitConversion.intBitsToFloat(
1698                          Math.min(Math.max((int)(H * 255.999f), 0), 255)
1699                        | Math.min(Math.max((int)(S * 255.999f), 0), 255) << 8
1700                        | Math.min(Math.max((int)(L * 255.999f), 0), 255) << 16
1701                        | (((int) (as + change * (ae - as)) & 0xFE) << 24));
1702    }
