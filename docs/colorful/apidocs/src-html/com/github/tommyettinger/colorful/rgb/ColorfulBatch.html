015/**
016 * A substitute for {@link com.badlogic.gdx.graphics.g2d.SpriteBatch} that adds an additional attribute to store an
017 * extra color's worth of channels, used to modify the RGB channels of a color by multiplication (called the "tweak")
018 * while the primary color affects the color additively (just called the color, often drawn from {@link Palette} or
019 * generated with {@link ColorTools}). This ColorfulBatch uses the RGB color space, like SpriteBatch, but the batch
020 * color here is additive for the red, green, and blue channels, while the tweak is multiplicative, and the neutral
021 * value for all RGB channels is 0.5 (instead of 1.0 for SpriteBatch). Additive values cause no change when they are
022 * 0.5, but cause a sharp increase at 1.0 and a sharp decrease at 0.0. In the tweak, r maps to multiplicative red,
023 * g maps to multiplicative green, b maps to multiplicative blue, and a maps to "exponential-like" contrast. Like with
024 * the additive values, 0.5 causes no change for multiplicative values, but you can think of the existing color as being
025 * temporarily shifted down by 0.5 for each of the RGB channels, so each RGB channel's range is -0.5 to 0.5 before being
026 * multiplied by the corresponding tweak value (times 2.0, which means multiplicative tweak values less than 0.5 make
027 * colors darker and values greater than 0.5 make colors lighter) and then shifted back to its prior range. Contrast is
028 * a special value; 0.5 still means "no change," but lower values will make lightness changes less distinct between
029 * similar colors, while higher values will "sharpen" the changes in lightness.
030 * <br>
031 * Created by Tommy Ettinger on 12/14/2019.
032 */

092    /** Constructs a new ColorfulBatch. Sets the projection matrix to an orthographic projection with y-axis point upwards, x-axis
093     * point to the right and the origin being in the bottom left corner of the screen. The projection will be pixel perfect with
094     * respect to the current screen resolution.
095     * <p>
096     * The defaultShader specifies the shader to use. Note that the names for uniforms for this default shader are different than
097     * the ones expect for shaders set with {@link #setShader(ShaderProgram)}. See {@link #createDefaultShader()}.
098     * @param size The max number of sprites in a single batch. Max of 10922.
099     * @param defaultShader The default shader to use. This is not owned by the ColorfulBatch and must be disposed separately. */
100    public ColorfulBatch(int size, ShaderProgram defaultShader) {
101        // 65535 is max vertex index, so 65535 / 6 vertices per sprite = 10922 sprites max.
102        if (size > 10922) throw new IllegalArgumentException("Can't have more than 10922 sprites per batch: " + size);
103
104        Mesh.VertexDataType vertexDataType = (Gdx.gl30 != null) ? Mesh.VertexDataType.VertexBufferObjectWithVAO : Mesh.VertexDataType.VertexArray;
105
106        mesh = new Mesh(vertexDataType, false, size * 4, size * 6,
107                new VertexAttribute(VertexAttributes.Usage.Position, 2, ShaderProgram.POSITION_ATTRIBUTE),
108                new VertexAttribute(VertexAttributes.Usage.ColorPacked, 4, ShaderProgram.COLOR_ATTRIBUTE),
109                new VertexAttribute(VertexAttributes.Usage.TextureCoordinates, 2, ShaderProgram.TEXCOORD_ATTRIBUTE + "0"),
110                new VertexAttribute(VertexAttributes.Usage.ColorPacked, 4, TWEAK_ATTRIBUTE));
111
112        projectionMatrix.setToOrtho2D(0, 0, Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
113
114        vertices = new float[size * SPRITE_SIZE];
115
116        int len = size * 6;
117        short[] indices = new short[len];
118        short j = 0;
119        for (int i = 0; i < len; i += 6, j += 4) {
120            indices[i] = j;
121            indices[i + 1] = (short)(j + 1);
122            indices[i + 2] = (short)(j + 2);
123            indices[i + 3] = (short)(j + 2);
124            indices[i + 4] = (short)(j + 3);
125            indices[i + 5] = j;
126        }
127        mesh.setIndices(indices);
128
129        if (defaultShader == null) {
130            shader = createDefaultShader();
131            ownsShader = true;
132        } else
133            shader = defaultShader;
134    }

136    /**
137     * Makes a new instance of the default ShaderProgram used for this ColorfulBatch, without any {@code #version}
138     * specified in the shader source. This expects an extra attribute (relative to a normal SpriteBatch) that is used
139     * for the tweak. You may want to set the code to prepend before you call this, as with:
140     * {@code ShaderProgram.prependVertexCode = "#version 110\n";
141     * ShaderProgram.prependFragmentCode = "#version 110\n";}
142     * The actual version can be different, and may need to be different for compatibility with some hardware.
143     * @return a new instance of the default shader used by ColorfulBatch for GL2 when no shader is specified
144     */
145    public static ShaderProgram createDefaultShader () {
146        String vertexShader = "attribute vec4 " + ShaderProgram.POSITION_ATTRIBUTE + ";\n"
147                + "attribute vec4 " + ShaderProgram.COLOR_ATTRIBUTE + ";\n"
148                + "attribute vec2 " + ShaderProgram.TEXCOORD_ATTRIBUTE + "0;\n"
149                + "attribute vec4 " + TWEAK_ATTRIBUTE + ";\n"
150                + "uniform mat4 u_projTrans;\n"
151                + "varying vec4 v_color;\n"
152                + "varying vec4 v_tweak;\n"
153                + "varying vec2 v_texCoords;\n"
154                + "\n"
155                + "void main()\n"
156                + "{\n"
157                + "   v_color = " + ShaderProgram.COLOR_ATTRIBUTE + ";\n"
158                + "   v_color.rgb = v_color.rgb * 2.0 - 1.0;\n"
159//                + "   v_color.rgb = v_color.rgb * 0.5 - 0.25;\n"
160                + "   v_color.a = v_color.a * (255.0/254.0);\n"
161                + "   v_tweak = " + TWEAK_ATTRIBUTE + ";\n"
162                + "   v_tweak.a = v_tweak.a * (255.0/254.0);\n"
163                + "   v_texCoords = " + ShaderProgram.TEXCOORD_ATTRIBUTE + "0;\n"
164                + "   gl_Position =  u_projTrans * " + ShaderProgram.POSITION_ATTRIBUTE + ";\n"
165                + "}\n";
166        String fragmentShader =
167                "#ifdef GL_ES\n" +
168                        "#define LOWP lowp\n" +
169                        "precision mediump float;\n" +
170                        "#else\n" +
171                        "#define LOWP \n" +
172                        "#endif\n" +
173                        "varying vec2 v_texCoords;\n" +
174                        "varying LOWP vec4 v_color;\n" +
175                        "varying LOWP vec4 v_tweak;\n" +
176                        "uniform sampler2D u_texture;\n" +
177                        "vec3 barronSpline(vec3 x, float shape) {\n" +
178                        "        const float turning = 0.5;\n" +
179                        "        vec3 d = turning - x;\n" +
180                        "        return mix(\n" +
181                        "          ((1. - turning) * (x - 1.)) / (1. - (x + shape * d)) + 1.,\n" +
182                        "          (turning * x) / (1.0e-20 + (x + shape * d)),\n" +
183                        "          step(0.0, d));\n" +
184                        "}\n" +
185                        "void main()\n" +
186                        "{\n" +
187                        "  vec4 tgt = texture2D( u_texture, v_texCoords );\n" +
188                        "  tgt.rgb = barronSpline(clamp(tgt.rgb * v_tweak.rgb * 2.0 + v_color.rgb, 0.0, 1.0), v_tweak.a * 1.5 + 0.25);\n" +
189                        "  tgt.a *= v_color.a;\n" +
190                        "  gl_FragColor = tgt;\n" +
191                        "}";
192        ShaderProgram shader = new ShaderProgram(vertexShader, fragmentShader);
193        if (!shader.isCompiled()) throw new IllegalArgumentException("Error compiling shader: " + shader.getLog());
194        return shader;
195    }

252    /**
253     * Sets the color with the given RGB and A parameters from 0 to 255.
254     * For the RGB parameters, 127 is a neutral value (causing no change), while for alpha, 255 is a neutral
255     * value. For RGB, higher values will add to the corresponding channel, while lower values will subtract from it.
256     * @see ColorTools#rgb(float, float, float, float)
257     * @param red additive red channel; ranges from 0 to 255, and 127 is neutral
258     * @param green additive green channel; ranges from 0 to 255, and 127 is neutral
259     * @param blue additive blue channel; ranges from 0 to 255, and 127 is neutral
260     * @param alpha multiplicative opacity; ranges from 0 to 255 (254 is equivalent, since the lowest bit is discarded)
261     */
262    public void setIntColor(int red, int green, int blue, int alpha) {
263        color = NumberUtils.intBitsToFloat((alpha << 24 & 0xFE000000)
264                | (blue << 16 & 0xFF0000) | (green << 8 & 0xFF00) | (red & 0xFF));
265    }

273    public Color getColor () {
274        final int intBits = NumberUtils.floatToRawIntBits(color);
275        Color color = tempColor;
276        color.r = (intBits & 0xff) / 255f;
277        color.g = ((intBits >>> 8) & 0xff) / 255f;
278        color.b = ((intBits >>> 16) & 0xff) / 255f;
279        color.a = ((intBits >>> 24) & 0xff) / 255f;
280        return color;
281    }

314    /**
315     * Takes the tweak as an int in the format: red (8 bits), green (8 bits), blue (8 bits), contrast (7 bits),
316     * (1 ignored bit at the end). An example would be 0xC820206E, which significantly emphasizes red (with red 0xC8,
317     * a little over 3/4 of the max and higher than the neutral value of 0x80), significantly reduces green and blue
318     * effect with green and blue multipliers of 0x20 (closer to 0, so most green and blue in the final color, if any,
319     * will come from the additive color), and slightly reduces contrast (with contrast and the ignored bit as 0x6E,
320     * which is less than the halfway point).
321     * @param tweak the tweak to use as an integer, with red in the most significant bits and contrast in least
322     */
323    public void setIntTweak(final int tweak) {
324        this.tweak = NumberUtils.intBitsToFloat(Integer.reverseBytes(tweak & -2));
325    }

371    @Override
372    public void draw (Texture texture, float x, float y, float originX, float originY, float width, float height, float scaleX,
373                      float scaleY, float rotation, int srcX, int srcY, int srcWidth, int srcHeight, boolean flipX, boolean flipY) {
374        if (!drawing) throw new IllegalStateException("ColorfulBatch.begin must be called before draw.");
375
376        float[] vertices = this.vertices;
377
378        if (texture != lastTexture)
379            switchTexture(texture);
380        else if (idx == vertices.length)
381            flush();
382
383        // bottom left and top right corner points relative to origin
384        final float worldOriginX = x + originX;
385        final float worldOriginY = y + originY;
386        float fx = -originX;
387        float fy = -originY;
388        float fx2 = width - originX;
389        float fy2 = height - originY;
390
391        // scale
392        if (scaleX != 1 || scaleY != 1) {
393            fx *= scaleX;
394            fy *= scaleY;
395            fx2 *= scaleX;
396            fy2 *= scaleY;
397        }
398
399        // construct corner points, start from top left and go counter clockwise
400        final float p1x = fx;
401        final float p1y = fy;
402        final float p2x = fx;
403        final float p2y = fy2;
404        final float p3x = fx2;
405        final float p3y = fy2;
406        final float p4x = fx2;
407        final float p4y = fy;
408
409        float x1;
410        float y1;
411        float x2;
412        float y2;
413        float x3;
414        float y3;
415        float x4;
416        float y4;
417
418        // rotate
419        if (rotation != 0) {
420            final float cos = MathUtils.cosDeg(rotation);
421            final float sin = MathUtils.sinDeg(rotation);
422
423            x1 = cos * p1x - sin * p1y;
424            y1 = sin * p1x + cos * p1y;
425
426            x2 = cos * p2x - sin * p2y;
427            y2 = sin * p2x + cos * p2y;
428
429            x3 = cos * p3x - sin * p3y;
430            y3 = sin * p3x + cos * p3y;
431
432            x4 = x1 + (x3 - x2);
433            y4 = y3 - (y2 - y1);
434        } else {
435            x1 = p1x;
436            y1 = p1y;
437
438            x2 = p2x;
439            y2 = p2y;
440
441            x3 = p3x;
442            y3 = p3y;
443
444            x4 = p4x;
445            y4 = p4y;
446        }
447
448        x1 += worldOriginX;
449        y1 += worldOriginY;
450        x2 += worldOriginX;
451        y2 += worldOriginY;
452        x3 += worldOriginX;
453        y3 += worldOriginY;
454        x4 += worldOriginX;
455        y4 += worldOriginY;
456
457        float u = srcX * invTexWidth;
458        float v = (srcY + srcHeight) * invTexHeight;
459        float u2 = (srcX + srcWidth) * invTexWidth;
460        float v2 = srcY * invTexHeight;
461
462        if (flipX) {
463            float tmp = u;
464            u = u2;
465            u2 = tmp;
466        }
467
468        if (flipY) {
469            float tmp = v;
470            v = v2;
471            v2 = tmp;
472        }
473
474        final float color = this.color;
475        final float tweak = this.tweak;
476        final int idx = this.idx;
477        vertices[idx] = x1;
478        vertices[idx + 1] = y1;
479        vertices[idx + 2] = color;
480        vertices[idx + 3] = u;
481        vertices[idx + 4] = v;
482        vertices[idx + 5] = tweak;
483
484        vertices[idx + 6] = x2;
485        vertices[idx + 7] = y2;
486        vertices[idx + 8] = color;
487        vertices[idx + 9] = u;
488        vertices[idx + 10] = v2;
489        vertices[idx + 11] = tweak;
490
491        vertices[idx + 12] = x3;
492        vertices[idx + 13] = y3;
493        vertices[idx + 14] = color;
494        vertices[idx + 15] = u2;
495        vertices[idx + 16] = v2;
496        vertices[idx + 17] = tweak;
497
498        vertices[idx + 18] = x4;
499        vertices[idx + 19] = y4;
500        vertices[idx + 20] = color;
501        vertices[idx + 21] = u2;
502        vertices[idx + 22] = v;
503        vertices[idx + 23] = tweak;
504        this.idx = idx + 24;
505    }

725    /**
726     * This is very different from the other overloads in this class; it assumes the float array it is given is in the
727     * format libGDX uses to give to SpriteBatch, that is, in groups of 20 floats per sprite. ColorfulBatch uses 24
728     * floats per sprite, to add tweak per color, so this does some conversion.
729     * @param texture the Texture being drawn from; usually an atlas or some parent Texture with lots of TextureRegions
730     * @param spriteVertices not the same format as {@link #vertices} in this class; should have a length that's a multiple of 20
731     * @param offset where to start drawing vertices from {@code spriteVertices}
732     * @param count how many vertices to draw from {@code spriteVertices} (20 vertices is one sprite)
733     */
734    @Override
735    public void draw (Texture texture, float[] spriteVertices, int offset, int count) {
736        if (!drawing) throw new IllegalStateException("ColorfulBatch.begin must be called before draw.");
737
738        count = (count / 5) * 6;
739        int verticesLength = vertices.length;
740        int remainingVertices = verticesLength;
741        if (texture != lastTexture)
742            switchTexture(texture);
743        else {
744            remainingVertices -= idx;
745            if (remainingVertices == 0) {
746                flush();
747                remainingVertices = verticesLength;
748            }
749        }
750        int copyCount = Math.min(remainingVertices, count);
751        final float tweak = this.tweak;
752
753        ////old way, breaks when libGDX code expects SPRITE_SIZE to be 20
754        //System.arraycopy(spriteVertices, offset, vertices, idx, copyCount);
755        ////new way, thanks mgsx
756        for (int s = offset, v = idx, i = 0; i < copyCount; i += 6) {
757            vertices[v++] = spriteVertices[s++];
758            vertices[v++] = spriteVertices[s++];
759            vertices[v++] = spriteVertices[s++];
760            vertices[v++] = spriteVertices[s++];
761            vertices[v++] = spriteVertices[s++];
762            vertices[v++] = tweak;
763        }
764        idx += copyCount;
765        count -= copyCount;
766        while (count > 0) {
767            offset += (copyCount / 6) * 5;
768            flush();
769            copyCount = Math.min(verticesLength, count);
770            ////old way, breaks when libGDX code expects SPRITE_SIZE to be 20
771            //System.arraycopy(spriteVertices, offset, vertices, 0, copyCount);
772            ////new way, thanks mgsx
773            for (int s = offset, v = 0, i = 0; i < copyCount; i += 6) {
774                vertices[v++] = spriteVertices[s++];
775                vertices[v++] = spriteVertices[s++];
776                vertices[v++] = spriteVertices[s++];
777                vertices[v++] = spriteVertices[s++];
778                vertices[v++] = spriteVertices[s++];
779                vertices[v++] = tweak;
780            }
781            idx += copyCount;
782            count -= copyCount;
783        }
784    }

786    /**
787     * Meant for code that uses ColorfulBatch specifically and can set an extra float (for the color tweak) per vertex,
788     * this is just like {@link #draw(Texture, float[], int, int)} when used in other Batch implementations, but expects
789     * {@code spriteVertices} to have a length that is a multiple of 24 instead of 20.
790     * @param texture the Texture being drawn from; usually an atlas or some parent Texture with lots of TextureRegions
791     * @param spriteVertices vertices formatted as this class uses them; length should be a multiple of 24
792     * @param offset where to start drawing vertices from {@code spriteVertices}
793     * @param count how many vertices to draw from {@code spriteVertices} (24 vertices is one sprite)
794     */
795    public void drawExactly (Texture texture, float[] spriteVertices, int offset, int count) {
796        if (!drawing) throw new IllegalStateException("ColorfulBatch.begin must be called before draw.");
797
798        int verticesLength = vertices.length;
799        int remainingVertices = verticesLength;
800        if (texture != lastTexture)
801            switchTexture(texture);
802        else {
803            remainingVertices -= idx;
804            if (remainingVertices == 0) {
805                flush();
806                remainingVertices = verticesLength;
807            }
808        }
809        int copyCount = Math.min(remainingVertices, count);
810
811        System.arraycopy(spriteVertices, offset, vertices, idx, copyCount);
812        idx += copyCount;
813        count -= copyCount;
814        while (count > 0) {
815            offset += copyCount;
816            flush();
817            copyCount = Math.min(verticesLength, count);
818            System.arraycopy(spriteVertices, offset, vertices, 0, copyCount);
819            idx += copyCount;
820            count -= copyCount;
821        }
822    }

1208    @SuppressWarnings("RedundantCast") // These casts are absolutely not redundant! Java 9 changed Buffer ABI.
1209    @Override
1210    public void flush () {
1211        if (idx == 0) return;
1212
1213        renderCalls++;
1214        totalRenderCalls++;
1215        int spritesInBatch = idx / SPRITE_SIZE;
1216        if (spritesInBatch > maxSpritesInBatch) maxSpritesInBatch = spritesInBatch;
1217        int count = spritesInBatch * 6;
1218
1219        lastTexture.bind();
1220        Mesh mesh = this.mesh;
1221        mesh.setVertices(vertices, 0, idx);
1222        ((Buffer)mesh.getIndicesBuffer()).position(0);
1223        ((Buffer)mesh.getIndicesBuffer()).limit(count);
1224
1225        if (blendingDisabled) {
1226            Gdx.gl.glDisable(GL20.GL_BLEND);
1227        } else {
1228            Gdx.gl.glEnable(GL20.GL_BLEND);
1229            if (blendSrcFunc != -1) Gdx.gl.glBlendFuncSeparate(blendSrcFunc, blendDstFunc, blendSrcFuncAlpha, blendDstFuncAlpha);
1230        }
1231
1232        mesh.render(customShader != null ? customShader : shader, GL20.GL_TRIANGLES, 0, count);
1233
1234        idx = 0;
1235    }
